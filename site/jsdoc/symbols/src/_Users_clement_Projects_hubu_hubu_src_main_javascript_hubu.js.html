 64     /**
 65      * Checks if the given component implements the
 66      * 'component' protocol (i.e.e interface).
 67      * @param {DE_AKQUINET.AbstractComponent} component the component to check
 68      * @return true if this is a valid component,
 69      * false otherwise.
 70      * @private
 71      */
 72     function checkComponent(component) {
 73         // if component is null, return false
 74         if (! component) {
 75             return false;
 76         }
 77 
 78         return DE_AKQUINET.utils
 79             .isObjectConformToContract(component,
 80                 new DE_AKQUINET.AbstractComponent());
 81     }

 84     /**
 85      * Processes the given event sent by the given component.
 86      * This methods checks who is interested by the event by
 87      * calling the match method, then the callback method is
 88      * called.
 89      * The event is extended with the 'source' property indicating
 90      * the component sending the event.
 91      * @param {Object} event
 92      * @param {DE_AKQUINET.AbstractComponent} component
 93      * @return true if the event was consume by at least one
 94      * component, false otherwise.
 95      * @private
 96      */
 97     function processEvent(component, event) {
 98         if (! event || ! component) {
 99             return false;
100         }
101 
102         var sent = false;
103 
104         for (var i = 0; i <listeners.length; i++) {
105             var listener = listeners[i];
106             // Skip the sender
107             if (listener.component == component) {
108                 continue;
109             }
110 
111             // Check match method
112 			// We clone the event to avoid modification impact
113             var ev = DE_AKQUINET.utils.clone(event);
114             ev.source = component;
115 
116             if (listener.match.apply(listener.component, [ev])) {
117                 listener.callback.apply(listener.component, [ev]);
118                 sent = true;
119             }
120         }
121 
122         return sent;
123     }

131         /**

140         /**
141          * Looks for one specific component plugged to the hub.
142          * This lookup is based on the component 'getComponentName'
143          * method.
144          * @param {String} name the component name
145          * @return the component with the matching name or 'null'
146          * if the component is not plugged.
147          */
148         getComponent : function(name) {
149             // If name is null, return null
150             if (! name) {
151                 return null;
152             }
153 
154             for (var i = 0; i < components.length; i++) {
155                 var cmp = components[i];
156                 // Check that we have the getComponentName function
157                 var fc = cmp.getComponentName;
158                 if (DE_AKQUINET.utils.isFunction(fc)) {
159                     var n = fc.apply(cmp, []); // Invoke the method.
160                     if (name == n) {
161                         // Only on match, just return.
162                         return cmp;
163                     }
164                 }
165             }
166             // Not found
167             return null;
168         },
169 
170         /**
171          * Registers a new component on the hub.
172          * If the component already exists, the registration is ignored. The lookup
173          * is based on the 'getComponentName' method.
174          * This method allows to configure the component.Once successfully registered,
175          * the hub call the 'configure' method on the component passing a reference
176          * on the hub and the configuration to the component.
177          * If component is null, the method throws an exception.
178          * @param {DE_AKQUINET.AbstractComponent} component the component to register
179          * @param {Object} configuration the component configuration (optional).
180          * If the configuration contain the 'component_name' key, the component
181          * takes this name.
182          * @return the current hub
183          */
184         registerComponent : function(component, configuration) {
185             if (!component) {
186                 throw "Cannot register component - component is null";
187             }
188 
189             // Check the validity of the component.
190             if (! checkComponent(component)) {
191                 if (component.getComponentName) {
192                     throw component.getComponentName() + " is not a valid component";
193                 } else {
194                     throw component + " is not a valid component";
195                 }
196             }
197 
198             // First check that we don't have already this component
199             // We can call getComponentName as we have check the component
200             if (this.getComponent(component.getComponentName())) {
201                 // If the component is already plugged, we return immediately
202                 return hub;
203             }
204 
205             // Add the component at the end of the list.
206 
207            (this.getComponents()).push(component);
208 
209            // Manage component_name
210            if (configuration && configuration.component_name) {
211                // Set a field containing the name
212                component.__name__ = configuration.component_name;
213                // Replace the method.
214                component.getComponentName = function() {
215                    return this.__name__;
216                };
217            }
218 
219            // Call configure on the component
220            // We pass the current hub
221            component.configure(this, configuration);
222 
223            // If we're already started, call start
224            if (started) {
225                component.start();
226            }
227 
228            // Return the current hub
229            return this;
230         },
231 
232         /**
233          * Unregisters the given component.
234          * If the component is not plugged to the hub, this method does nothing.
235          * @param {Object} component either the component object ({DE_AKQUINET.AbstractComponent})
236          * or the component name {String}
237          * @return the current hub.
238          */
239         unregisterComponent : function(component) {
240             // Check parameter
241             var cmp;
242 
243             // If component is null, return immediately
244             if (! component) {
245                 return this;
246             }
247 
248 
249             if (typeof component === "string") {
250                cmp = this.getComponent(component);
251                // If the component is not plugged,
252                // Exit immediately
253                if (! cmp) {
254                    return this;
255                }
256             } else {
257                 if (! checkComponent(component)) {
258                     throw component + " is not a valid component";
259                 } else {
260                     cmp = component;
261                 }
262             }
263 
264             // Iterate on the components array to find the component to
265             // unregister.
266             var idx = DE_AKQUINET.utils.indexOf(components, cmp); // Find the index
267             if (idx != -1) { // Remove it if really found
268                 // Call stop on the component
269                 cmp.stop();
270                 this.unregisterListener(cmp);
271                 components.splice(idx, 1);
272             }
273 
274             return this;
275         },

277         /**
278          * Starts the hub.
279          * This method calls start on all plugged components.
280          * This method does nothing is the hub is already started.
281          * @return the hub
282          */
283         start : function() {
284             if (started) {
285                 return this;
286             }
287             started = true;
288             for (var i = 0; i < components.length; i++) {
289                 // Only valid component can be plugged
290                 // So we can call start directly.
291                 components[i].start();
292             }
293             return this;
294         },

296         /**
297          * Stops the hub.
298          * This method calls stop on all plugged components.
299          * If the hub is not started, this methods does nothing.
300          * @return the hub
301          */
302         stop : function() {
303             if (! started ) {
304                 return this;
305             }
306             started = false;
307             for (var i = 0; i < components.length; i++) {
308                 // Only valid component can be plugged
309                 // So we can call stop directly.
310                 components[i].stop();
311             }
312             return this;
313         },

315         /**
316          * Binds a component to another component.
317          * @param {Object} binding this object describes the binding.
318          * It contains:
319          * <ul>
320          * <li>component : the component to inject ({String} or {DE_AKQUINET.AbstractComponent} object)</li>
321          * <li>to : the component receiving the 'component' ({String} or {Component}DE_AKQUINET.AbstractComponent} object)</li>
322          * <li>into : where the component is injected, it can be either a function (called on 'to' with 'component' as parameter),
323          * or a {String}. In this latter case,'to'.'into' is injected. If it's a function the function is called, if it's an object
324          * it is just injected (assigned).</li>
325          * <li>contract</li> : The binding contract (i.e. interface) (optional). The contract is a simple object defining the methods
326          * that the destination can invoke on the source. If the source is not conform to the contract, the binding is rejected. The
327          * contract is a kind of interface. By default a proxy implementing the contract is injected enforcing the contract-based
328 		 * interaction. However, you can disable the proxy by setting <tt>proxy:false</tt></li>
329 		 * <li>proxy</li> : boolean indicating if proxies are enabled or disabled. By default they are enabled if <tt>contract</tt> is
330 		 * set, however, setting proxy to false enables the injection of a diract reference.
331          * @return the current hub
332          */
333         bind : function(binding) {
334             // First check that all parameters are here
335             if (! binding.component  || ! binding.to || ! binding.into) {
336                 throw "Cannot bind components - component, to and into must be defined";
337             }
338 
339             // Get the first component object
340             var component;
341             // Two cases, either component is a String, or an Object
342             if (typeof binding.component === "string") {
343                component = this.getComponent(binding.component);
344                // If the component is not plugged, i.e. lookup failed
345                if (! component) {
346                    throw "Cannot bind components - 'component' is not plugged to the hub";
347                }
348             } else { // It's a component
349                 if (! checkComponent(binding.component)) {
350                     throw binding.component + " is not a valid component";
351                 } else {
352                     component = binding.component;
353                 }
354             }
355 
356             if (binding.contract) {
357                 // A contract is set, check that the component is confform to the contract
358                 // We cannot do this checking before because the component was unknown
359                 if (! DE_AKQUINET.utils.isObjectConformToContract(component, binding.contract)) {
360                     throw "Cannot bind components - 'component' is not conform to the contract";
361                 } else {
362 					// Do we have to create a proxy ?
363 					//TODO This test can be better.
364 					if (binding.proxy == null  || binding.proxy) {
365 						// Create the proxy
366 						component = DE_AKQUINET.utils.createProxyForContract(binding.contract, component);
367 					} else {
368 						// Direct injection
369 						// component = component so nothing to do.
370 					}
371 				}
372             }
373 
374             // Get the second component (to)
375             var to; // The computed component
376             // Two cases, either component is a String, or an Object
377             if (typeof binding.to === "string") {
378                to = this.getComponent(binding.to);
379                // If the component is not plugged,
380                if (! to) {
381                    throw "Cannot bind components - 'to' is not plugged to the hub";
382                }
383             } else { // It's a component
384                 if (! checkComponent(binding.to)) {
385                     throw binding.to + " is not a valid component";
386                 } else {
387                     to = binding.to;
388                 }
389             }
390 
391             // Detect injection mechanism
392             // First case, into is a function
393             if (DE_AKQUINET.utils.isFunction(binding.into)) {
394                 // We call the function on the 'to' object
395                 // We pass 'component' as parameter
396                 binding.into.apply(to, [component]);
397             } else if (typeof binding.into === "string") {
398                 // Lookup the member
399                 var injector = to[binding.into];
400                 if (! injector) {
401                     // Just create the member and inject it
402                     to[binding.into] = component;
403                 } else {
404                     // injector already exist
405                     // if it's a function call the function otherwise
406                     // assign the value
407                     if (DE_AKQUINET.utils.isFunction(injector)) {
408                         injector.apply(to, [component]);
409                     } else {
410                         to[binding.into] = component;
411                     }
412                 }
413             } else {
414                 // Unsupported case
415                 throw ("Cannot bind components = 'into' must be either a function or a string");
416             }
417 
418             return this;
419         },

430         /**
431          * Registers a listener.
432          * @param {DE_AKQUINET.AbstractComponent} component : the
433          * component registering the listener
434          * @param {Function} match : the method called to check if the event
435          * matches. This method must returns true or false.
436          * @param {Function} callback : the callback method to invoke when
437          * a matching event is sent
438          */
439         registerListener : function(component, match, callback) {
440           // Nothing can be null
441           if (! component || ! callback || ! match) {
442               throw "Cannot register the listener - all parameters must be defined";
443           }
444 
445           // check that the component is plugged to the hub
446           // we don't have to check if the component is valid, because
447           // only valid components can be plugged to the hub.
448           var idx = DE_AKQUINET.utils.indexOf(components, component);
449           if (idx == -1) {
450               throw "Cannot register the listener - The component is not plugged to the hub";
451           }
452 
453           // Add the lister to the listener list:
454           // Create the object
455           var listener = {
456               'component': component,
457               'callback' : callback,
458               'match' : match
459           };
460           // Add the object at the end of the listener array
461           listeners.push(listener);
462           return this;
463         },

484               throw "Cannot register the listener - The component is not plugged to the hub";
485           }

488           // Create the object

490               'component': component,
491               'callback' : callback,
492               'match' : match
493           };

500         /**
501          * Unregisters listeners for the given component.
502          * According to the arguments, several cases
503          * can happen:
504          * 'component' can be either a String of a Component.
505          * In case of the string, we look up for the component using
506          * 'getComponentName'.
507          * If 'callback' is defined, only the listener matching component
508          * and callback will be unregistered. Otherwise all listeners of the
509          * component will be unregistered.
510          * If 'component' is null, this methods does nothing.
511          * @param {DE_AKQUINET.AbstractComponent} component the component
512          * @param {Function} callback the callback function (optional)
513          * @return the current hub
514          *
515          */
516         unregisterListener : function(component, callback) {
517             // Check that we have at least a correct component
518             if (! component) {
519                 return this;
520             }
521 
522             var cmp; // The computed component
523             // Two cases, either component is a String, or an Object
524             if (typeof component === "string") {
525                cmp = this.getComponent(component);
526                // If the component is not plugged,
527                // Exit immediately
528                if (! cmp) {
529                    return this;
530                }
531             } else { // It's a component
532                 if (! checkComponent(component)) {
533                     throw component + " is not a valid component";
534                 } else {
535                     cmp = component;
536                 }
537             }
538 
539             // So, here cmp is the component.
540             var toRemove = [];
541 			var i; // Loop index;
542 			var listener;
543             if (callback) {
544                 // Must lookup component and callback
545                 for (i = 0; i < listeners.length; i++) {
546                     listener = listeners[i];
547                     if (listener.component == cmp  && listener.callback == callback) {
548                         // Match
549                         toRemove.push(listener);
550                     }
551                 }
552             } else {
553                 for (i = 0; i < listeners.length; i++) {
554                     listener = listeners[i];
555                     if (listener.component == cmp) {
556                         // Match
557                         toRemove.push(listener);
558                     }
559                 }
560             }
561 
562             // We must remove all listeners contained in toRemove
563             for (i = 0; i < toRemove.length; i++) {
564                 var idx = DE_AKQUINET.utils.indexOf(listeners, toRemove[i]); // Find the index
565                 if (idx != -1) { // Remove it if really found, that should always be the case.
566                     listeners.splice(idx, 1);
567                 }
568             }
569         },

573          * If component or event is null, the method
574          * does nothing. If not, the event processed
575          * and sent to all matching listeners.
576          * @param {DE_AKQUINET.AbstractComponent} component

583             if (! component || ! event) {
584                 return false;

686 /**
687  * Checks that the given object is conform to the given contract
688  * The contract is a javascript object.
689  * The conformity is computed as follow:
690  * <code>
691  * O is conform to C if and only if for all i in C where C[i] != null
692  * O[i] != null && typeof(C[i]) = typeOf(O[i])
693  * </code>
694  * This is an implementation of 'Chi':
695  * <code>Metamodel <- chi <- Model -> mu -> System</code>
696  * where chi is : isObjectConformToContract and mu is representationOf.
697  * @param object the object to check
698  * @param contract the contract
699  * @return true if the object is conform with the given contract, false
700  * otherwise.
701  */
702 DE_AKQUINET.utils.isObjectConformToContract = function(object, contract) {
703     // For all 'properties' from contract, check that the object
704     // has an equivalent property
705     for (var i in contract) {
706         if (object[i] === null) {
707             return false;
708         } else {
709             // Check type
710             if (typeof(contract[i]) != typeof (object[i])) {
711                 return false;
712             }
713         }
714     }
715     return true;
716 };

719 /**
720  * Utility method to check if the given object
721  * is a function. This code comes from JQuery.
722  * @param {Object} obj the object to test
723  * @returns true if the given object is a function
724  */
725 DE_AKQUINET.utils.isFunction = function(obj) {
726   // We need to specify the exact function
727   // because toString can be overriden by browser.
728   return Object.prototype.toString.call(obj) === "[object Function]";
729 };

731 /**
732  * Clones the given object. This create a deep copy of the
733  * given object.
734  * @param {Object} object the object to clone
735  * @return the cloned object
736  */
737 DE_AKQUINET.utils.clone = function(object) {
738     var newObj = (object instanceof Array) ? [] : {};
739     for (var i in object) {
740         if (object[i] && typeof object[i] == "object") {
741             newObj[i] = clone(object[i]);
742         }
743         else {
744             newObj[i] = object[i];
745         }
746     }
747     return newObj;
748 };

750 /**
751  * Creates a proxy hiding the given object. The proxy
752  * implements the contract (and only the contract).
753  * @param {Object} contract the contract
754  * @param {Object} object the object to proxy
755  * @return the proxy
756  */
757 DE_AKQUINET.utils.createProxyForContract = function(contract, object) {
758 	var proxy = {};
759 	// We inject the proxied objects
760 	proxy.__proxy__ = object;
761 
762 	for (var i in contract) {
763 		if (this.isFunction(contract[i])) {
764 			// To call the correct method, we create a new anonymous function
765 			// applying the arguments on the function itself
766 			// We use apply to pass all arguments, and set the target
767 			// The resulting method is stored using the method name in the
768 			// proxy object
769 			// TODO This method uses 'eval', we should change !
770 			proxy[i] = new Function('return this.__proxy__.' + i + '.apply(this.__proxy__, arguments)');
771 		} else {
772 			// Everything else is just referenced.
773             proxy[i] = object[i];
774 		}
775 	}
776 
777 	return proxy;
778 };

780 /**
781  * indexOf function.
782  * This method delegates on Array.indexOf if
783  * it exists. If not (IE), it just implements
784  * its own indexof.
785  * @param {Object} array the array
786  * @param {Object} obj the object
787  * @return the index of the object 'obj' in the array or
788  * -1 if not found.
789  */
790 DE_AKQUINET.utils.indexOf = function(array, obj) {
791   if (Array.indexOf) {
792 	  // We just delegate hoping some native code (optimization)
793       return array.indexOf(obj);
794   } else {
795 	  // Simple lookup
796       for(var i = 0; i < array.length; i++){
797             if(array[i] == obj){
798                 return i;
799             }
800         }
801         return -1;
802   }
